#include <iostream>
using namespace std;
int compareStrings(const char *a, const char *b){
int i = 0;
while(a[i] != '\0' && b[i] != '\0'){
if(a[i] != b[i]) return a[i] - b[i];
i++;}
 return a[i] - b[i];}
void copyString(char *dest, const char *src){
int i = 0;
while(src[i] != '\0'){
dest[i] = src[i];
i++;}
dest[i] = '\0';}
int main()
{char arr[5][20] = {"banana", "apple", "cherry", "mango", "pear"};
char temp[20];
for(int i = 0; i < 5 - 1; i++){
for(int j = i + 1; j < 5; j++){
if(compareStrings(arr[i], arr[j]) > 0){
copyString(temp, arr[i]);
copyString(arr[i], arr[j]);
copyString(arr[j], temp);}}}
